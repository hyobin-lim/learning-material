자바스크립트 기초 학습 문서 (주석 포함)

이 문서는 자바스크립트의 기초 개념을 설명하고, 예제 코드와 데모를 함께 제공합니다. 각 코드에는 주석이 달려 있어 초보자도 쉽게 이해할 수 있습니다.

1. 변수와 출력

// demoVariable 이라는 이름의 함수를 정의합니다.
function demoVariable() {
  // let: 변수를 선언하는 키워드 (값을 바꿀 수 있음)
  let name = "효빈";   // 문자열(String) 타입
  let age = 25;        // 숫자(Number) 타입

  // document: 현재 웹페이지 전체를 가리킴
  // getElementById("demo1"): id="demo1"인 HTML 요소를 찾음
  // textContent: 그 요소 안에 글자를 넣음
  document.getElementById("demo1").textContent =
    "이름: " + name + ", 나이: " + age;
}

2. 함수와 이벤트

// demoFunction 이라는 함수를 정의합니다.
function demoFunction() {
  // greet 라는 이름의 함수를 내부에 정의합니다.
  function greet(user) {
    return "안녕하세요, " + user + "님!";
  }

  // id="demo2"인 HTML 요소에 greet("효빈") 결과를 넣음
  document.getElementById("demo2").textContent = greet("효빈");
}

3. DOM 제어

이 문장의 색상이 바뀝니다.

// changeColor 라는 함수를 정의합니다.
function changeColor() {
  // id="demo3"인 HTML 요소를 찾아서 text 변수에 저장
  const text = document.getElementById("demo3");

  // 삼항 연산자: 조건 ? 참일때 : 거짓일때
  // 현재 색상이 blue면 red로, 아니면 blue로 바꿈
  text.style.color = text.style.color === "blue" ? "red" : "blue";
}

4. 조건문과 반복문

function demo_flow() {
  const score = 82; // 점수 변수
  let grade;        // 학점 변수

  // if/else 조건문: 점수에 따라 학점 결정
  if (score >= 90) grade = "A";
  else if (score >= 80) grade = "B";
  else grade = "C";

  // for 반복문: 1부터 5까지 숫자를 배열에 넣기
  let nums = [];
  for (let i = 1; i <= 5; i++) {
    nums.push(i);
  }

  // while 반복문: count가 0이 될 때까지 반복
  let count = 3;
  let msg = "";
  while (count > 0) {
    msg += "카운트 " + count + " ";
    count--;
  }

  // 결과 출력
  document.getElementById("flowOut").textContent =
    `score=${score}, grade=${grade}, nums=${nums.join(", ")}, msg=${msg}`;
}

5. 배열과 메서드

function demo_arr() {
  const scores = [72, 88, 91, 60, 45]; // 점수 배열

  // map: 배열의 각 요소를 변환
  const plus5 = scores.map(s => s + 5);

  // filter: 조건에 맞는 요소만 남김
  const passed = plus5.filter(s => s >= 70);

  // reduce: 배열을 하나의 값으로 줄임
  const sum = passed.reduce((acc, cur) => acc + cur, 0);

  // 결과 출력
  document.getElementById("arrOut").textContent =
    `원본=${scores.join(", ")} | +5=${plus5.join(", ")} | 통과=${passed.join(", ")} | 합계=${sum}`;
}

6. 객체와 메서드

function demo_obj() {
  // 객체(Object): 관련 데이터와 기능을 묶은 구조
  const user = {
    name: "효빈",   // 속성 name
    age: 25,        // 속성 age
    greet() {       // 메서드 greet
      return `안녕, 나는 ${this.name}이고 나이는 ${this.age}야.`;
    }
  };

  // 구조 분해 할당: 객체에서 필요한 값만 꺼내오기
  const { name, age } = user;

  // 결과 출력
  document.getElementById("objOut").textContent =
    `${user.greet()} (구조분해: name=${name}, age=${age})`;
}
