자바스크립트 기초 학습 페이지

이 페이지는 조건문, 반복문, 배열, 객체 같은 기초 문법을 예제와 함께 설명합니다.

조건문 (if/else)

// demo_condition 함수: 점수에 따라 학점 출력
function demo_condition() {
  const score = 85; // 점수 변수
  let grade;

  // if/else 조건문: 점수에 따라 학점 결정
  if (score >= 90) grade = "A";
  else if (score >= 80) grade = "B";
  else grade = "C";

  document.getElementById("condOut").textContent =
    `점수=${score}, 학점=${grade}`;
}

반복문 (for/while)

// demo_loop 함수: for와 while 반복문 예제
function demo_loop() {
  // for 반복문: 1부터 5까지 숫자 출력
  let nums = [];
  for (let i = 1; i <= 5; i++) {
    nums.push(i);
  }

  // while 반복문: 카운트 다운
  let count = 3;
  let msg = "";
  while (count > 0) {
    msg += "카운트 " + count + " ";
    count--;
  }

  document.getElementById("loopOut").textContent =
    `for 결과=${nums.join(", ")} | while 결과=${msg}`;
}

배열과 메서드 (map, filter, reduce)

// demo_array 함수: 배열 메서드 활용
function demo_array() {
  const scores = [72, 88, 91, 60, 45]; // 점수 배열

  // map: 각 점수에 +5
  const plus5 = scores.map(s => s + 5);

  // filter: 70 이상만 통과
  const passed = plus5.filter(s => s >= 70);

  // reduce: 합계 계산
  const sum = passed.reduce((acc, cur) => acc + cur, 0);

  document.getElementById("arrOut").textContent =
    `원본=${scores.join(", ")} | +5=${plus5.join(", ")} | 통과=${passed.join(", ")} | 합계=${sum}`;
}

객체와 메서드

// demo_object 함수: 객체와 메서드 활용
function demo_object() {
  // 객체(Object): 관련 데이터와 기능을 묶은 구조
  const user = {
    name: "효빈",   // 속성 name
    age: 25,        // 속성 age
    greet() {       // 메서드 greet
      return `안녕, 나는 ${this.name}이고 나이는 ${this.age}야.`;
    }
  };

  // 구조 분해 할당: 객체에서 필요한 값만 꺼내오기
  const { name, age } = user;

  document.getElementById("objOut").textContent =
    `${user.greet()} (구조분해: name=${name}, age=${age})`;
}
